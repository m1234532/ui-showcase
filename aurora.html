Aurora Flowing Text

/* 极光背景基础结构 */
.aurora-bg {
  background:     radial-gradient(ellipse at 20% 20%, rgba(120, 119, 198, 0.8) 0%, transparent 50%),
    radial-gradient(ellipse at 80% 80%, rgba(255, 119, 198, 0.6) 0%, transparent 50%),
    linear-gradient(135deg, #1a1a2e 0%, #16213e 100%)   ;
  background-size: 200% 200%;
  animation: aurora-flow 15s ease infinite;
}

/* 流动动画 */
@keyframes aurora-flow {
  0%, 100% { background-position: 0% 50%; }
  50% { background-position: 100% 50%; }
}
